012package org.atteo.moonshine.atomikos;
013
014import java.lang.management.ManagementFactory;
015import java.util.Properties;
016
017import javax.inject.Singleton;
018import javax.jms.ConnectionFactory;
019import javax.jms.XAConnectionFactory;
020import javax.sql.DataSource;
021import javax.sql.XADataSource;
022import javax.transaction.SystemException;
023import javax.transaction.TransactionManager;
024import javax.transaction.UserTransaction;
025import javax.xml.bind.annotation.XmlElement;
026import javax.xml.bind.annotation.XmlRootElement;
027
028import org.atteo.evo.config.XmlDefaultValue;
029import org.atteo.moonshine.jta.JtaConnectionFactoryWrapper;
030import org.atteo.moonshine.jta.JtaDataSourceWrapper;
031import org.atteo.moonshine.jta.JtaService;
032import org.atteo.moonshine.jta.PoolOptions;
033
034import com.atomikos.icatch.SysException;
035import com.atomikos.icatch.config.UserTransactionServiceImp;
036import com.atomikos.icatch.config.imp.AbstractUserTransactionServiceFactory;
037import com.atomikos.icatch.jta.UserTransactionManager;
038import com.atomikos.icatch.standalone.UserTransactionServiceFactory;
039import com.atomikos.jdbc.AtomikosDataSourceBean;
040import com.atomikos.jms.AtomikosConnectionFactoryBean;
041import com.google.inject.AbstractModule;
042import com.google.inject.Module;
043
044/**
045 * Atomikos JTA implementation.
046 */
047@XmlRootElement(name = "atomikos")
048public class Atomikos extends JtaService {
049    /**
050     * Specifies the maximum number of active transactions.
051     * <p>
052     * A negative value means infinite amount. You will get an IllegalStateException
053     * with error message "Max number of active transactions reached" if you call
054     * {@link UserTransaction#begin()} while there are already n concurrent transactions running,
055     * n being this value.
056     * </p>
057     */
058    @XmlElement
059    private Integer maxActiveTransactions = -1;
060
061    @XmlElement
062    @XmlDefaultValue("${dataHome}/atomikos/logs")
063    private String logDirectory;
064
065    @XmlElement
066    @XmlDefaultValue("${dataHome}/atomikos/")
067    private String consoleOutputDirectory;
068
069    /**
070     * The default timeout (in seconds) that is set for transactions when no timeout is specified.
071     */
072    @XmlElement
073    @XmlDefaultValue("60")
074    private Integer transactionTimeout;
075
076    private UserTransactionManager manager;
077    private UserTransactionServiceImp service;
078
079    @Singleton
080    private static class AtomikosDataSourceWrapper implements JtaDataSourceWrapper {
081        @Override
082        public DataSource wrap(String name, XADataSource xaDataSource, PoolOptions poolOptions, String testQuery) {
083            AtomikosDataSourceBean wrapped = new AtomikosDataSourceBean();
084            wrapped.setXaDataSource(xaDataSource);
085            wrapped.setUniqueResourceName(name);
086            wrapped.setTestQuery(testQuery);
087            if (poolOptions == null) {
088                return wrapped;
089            }
090
091            if (poolOptions.getMinPoolSize() != null) {
092                wrapped.setMinPoolSize(poolOptions.getMinPoolSize());
093            }
094            if (poolOptions.getMaxPoolSize() != null) {
095                wrapped.setMaxPoolSize(poolOptions.getMaxPoolSize());
096            }
097            if (poolOptions.getMaxIdleTime() != null) {
098                wrapped.setMaxIdleTime(poolOptions.getMaxIdleTime());
099            }
100
101            if (poolOptions.getReapTimeout() != null) {
102                wrapped.setReapTimeout(poolOptions.getReapTimeout());
103            }
104            return wrapped;
105        }
106
107        @Override
108        public void close(DataSource dataSource) {
109            ((AtomikosDataSourceBean) dataSource).close();
110        }
111    }
112
113    @Singleton
114    private static class AtomikosConnectionFactoryWrapper implements JtaConnectionFactoryWrapper {
115        @Override
116        public ConnectionFactory wrap(String name, XAConnectionFactory xaFactory,
117                PoolOptions poolOptions) {
118            AtomikosConnectionFactoryBean wrapped = new AtomikosConnectionFactoryBean();
119            wrapped.setXaConnectionFactory(xaFactory);
120            wrapped.setUniqueResourceName(name);
121            if (poolOptions == null) {
122                return wrapped;
123            }
124
125            if (poolOptions.getMinPoolSize() != null) {
126                wrapped.setMinPoolSize(poolOptions.getMinPoolSize());
127            }
128            if (poolOptions.getMaxPoolSize() != null) {
129                wrapped.setMaxPoolSize(poolOptions.getMaxPoolSize());
130            }
131            if (poolOptions.getMaxIdleTime() != null) {
132                wrapped.setMaxIdleTime(poolOptions.getMaxIdleTime());
133            }
134
135            if (poolOptions.getReapTimeout() != null) {
136                wrapped.setReapTimeout(poolOptions.getReapTimeout());
137            }
138            return wrapped;
139        }
140
141        @Override
142        public void close(ConnectionFactory connectionFactory) {
143            ((AtomikosConnectionFactoryBean) connectionFactory).close();
144        }
145    }
146
147    @Override
148    public Module configure() {
149        return new AbstractModule() {
150            @Override
151            protected void configure() {
152                System.setProperty(UserTransactionServiceImp.NO_FILE_PROPERTY_NAME, "true");
153                System.setProperty(UserTransactionServiceImp.HIDE_INIT_FILE_PATH_PROPERTY_NAME,
154                        "true");
155                System.setProperty("com.atomikos.icatch.service",
156                        UserTransactionServiceFactory.class.getCanonicalName());
157
158                Properties properties = new Properties();
159                properties.setProperty(AbstractUserTransactionServiceFactory.MAX_ACTIVES_PROPERTY_NAME,
160                        Integer.toString(maxActiveTransactions));
161                properties.setProperty(AbstractUserTransactionServiceFactory.TM_UNIQUE_NAME_PROPERTY_NAME,
162                        "TM_" + ManagementFactory.getRuntimeMXBean().getName());
163                properties.setProperty(AbstractUserTransactionServiceFactory.LOG_BASE_NAME_PROPERTY_NAME,
164                        "log_");
165                properties.setProperty(AbstractUserTransactionServiceFactory.LOG_BASE_DIR_PROPERTY_NAME,
166                        logDirectory);
167                properties.setProperty(AbstractUserTransactionServiceFactory.OUTPUT_DIR_PROPERTY_NAME,
168                        consoleOutputDirectory);
169                properties.setProperty(AbstractUserTransactionServiceFactory.THREADED_2PC_PROPERTY_NAME,
170                        "false");
171                properties.setProperty(AbstractUserTransactionServiceFactory.DEFAULT_JTA_TIMEOUT_PROPERTY_NAME,
172                        Integer.toString(transactionTimeout * 1000));
173                service = new UserTransactionServiceImp(properties);
174                try {
175                    service.init();
176                } catch (SysException e) {
177                    throw new RuntimeException(e.getErrors().pop().toString(), e);
178                }
179
180                manager = new UserTransactionManager();
181                manager.setStartupTransactionService(false);
182                try {
183                    manager.init();
184                } catch (SystemException e) {
185                    throw new RuntimeException(e);
186                }
187                bind(TransactionManager.class).toInstance(manager);
188                bind(UserTransaction.class).toInstance(manager);
189                bind(JtaDataSourceWrapper.class).to(AtomikosDataSourceWrapper.class);
190                bind(JtaConnectionFactoryWrapper.class).to(AtomikosConnectionFactoryWrapper.class);
191            }
192        };
193    }
194
195    @Override
196    public void close() {
197        manager.close();
198        service.shutdownWait();
199    }
200}
