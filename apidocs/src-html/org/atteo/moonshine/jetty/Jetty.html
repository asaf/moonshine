014package org.atteo.moonshine.jetty;
015
016import java.util.Arrays;
017import java.util.HashMap;
018import java.util.Map;
019
020import javax.management.MBeanServer;
021import javax.servlet.Filter;
022import javax.servlet.annotation.WebFilter;
023import javax.servlet.annotation.WebInitParam;
024import javax.servlet.annotation.WebServlet;
025import javax.servlet.http.HttpServlet;
026import javax.xml.bind.annotation.XmlElement;
027import javax.xml.bind.annotation.XmlElementRef;
028import javax.xml.bind.annotation.XmlElementWrapper;
029import javax.xml.bind.annotation.XmlRootElement;
030
031import org.atteo.evo.classindex.ClassIndex;
032import org.atteo.moonshine.webserver.WebServerService;
033import org.eclipse.jetty.jmx.MBeanContainer;
034import org.eclipse.jetty.server.Server;
035import org.eclipse.jetty.util.log.Log;
036
037import com.google.inject.Inject;
038import com.google.inject.Module;
039import com.google.inject.Singleton;
040import com.google.inject.servlet.ServletModule;
041
042/**
043 * Starts embedded Jetty web server instance.
044 *
045 * <p>
046 * There are two ways to register servlets and filters:
047 * <ul>
048 * <li>by annotating {@link HttpServlet servlet} with &#064;{@link WebServlet}
049 * and {@link Filter filter} with &#064;{@link WebFilter}</li>
050 * <li>by registering {@link ServletModule} in Guice</li>
051 * </ul>
052 * </p>
053 * <p>
054 * Please note that when using annotations the {@link ClassIndex} facility
055 * is used to get the list of annotated classes as opposed to otherwise more common
056 * classpath scanning. The implication of that is that the class will be "visible"
057 * to this Jetty service only when it was compiled with webserver.jar in the classpath
058 * which contains annotation processor generating the needed class index.
059 * Use {@link ServletModule} when that is not the case.
060 * </p>
061 */
062@XmlRootElement(name = "jetty")
063public class Jetty extends WebServerService {
064    /**
065     * Support {@link WebServlet}, {@link WebFilter} and {@link WebInitParam} annotations
066     * by automatically registering annotated classes using {@link ServletModule}.
067     */
068    @XmlElement
069    private boolean registerAnnotatedServlets = true;
070
071    @XmlElementWrapper(name = "sslcontextfactories")
072    @XmlElementRef
073    private SslContextFactoryConfig[] sslContextFactories;
074
075
076    /**
077     * List of connectors.
078     */
079    @XmlElementWrapper(name = "connectors")
080    @XmlElementRef
081    private ConnectorConfig[] connectors = new ConnectorConfig[] {
082        new ServerConnectorConfig()
083    };
084
085    /**
086     * Main handler. Usually a {@link HandlerCollectionConfig collection} or {@link HandlerListConfig list}
087     * of handlers. By default set to {@link ServletContextHandlerConfig}.
088     */
089    @XmlElementRef
090    private HandlerConfig handler = new ServletContextHandlerConfig();
091
092    @Override
093    public Module configure() {
094        return new ServletModule() {
095            @Override
096            protected void configureServlets() {
097                for (ConnectorConfig connector : connectors) {
098                    Module module = connector.configure();
099                    if (module != null) {
100                        install(module);
101                    }
102                }
103
104                Module module = handler.configure();
105                if (module != null) {
106                    install(module);
107                }
108
109                if (!registerAnnotatedServlets) {
110                    return;
111                }
112                for (Class<?> klass : ClassIndex.getAnnotated(WebServlet.class)) {
113                    WebServlet annotation = klass.getAnnotation(WebServlet.class);
114                    String[] urls = annotation.value();
115                    if (urls.length == 0) {
116                        urls = annotation.urlPatterns();
117                    }
118
119                    Map<String, String> params = new HashMap<>();
120
121                    for (WebInitParam param : annotation.initParams()) {
122                        params.put(param.name(), param.value());
123                    }
124
125                    String[] restOfUrls = Arrays.copyOfRange(urls, 1, urls.length);
126                    @SuppressWarnings("unchecked")
127                    Class<? extends HttpServlet> servletClass = (Class<? extends HttpServlet>) klass;
128                    serve(urls[0], restOfUrls).with(servletClass, params);
129                    bind(servletClass).in(Singleton.class);
130                }
131                for (Class<?> klass : ClassIndex.getAnnotated(WebFilter.class)) {
132                    WebFilter annotation = klass.getAnnotation(WebFilter.class);
133                    String[] urls = annotation.value();
134                    if (urls.length == 0) {
135                        urls = annotation.urlPatterns();
136                    }
137
138                    Map<String, String> params = new HashMap<>();
139
140                    for (WebInitParam param : annotation.initParams()) {
141                        params.put(param.name(), param.value());
142                    }
143
144                    String[] restOfUrls = Arrays.copyOfRange(urls, 1, urls.length - 1);
145                    @SuppressWarnings("unchecked")
146                    Class<? extends Filter> filterClass = (Class<? extends Filter>) klass;
147                    filter(urls[0], restOfUrls).through(filterClass, params);
148                    bind(filterClass).in(Singleton.class);
149                }
150            }
151        };
152    }
153
154    @Inject(optional = true)
155    private MBeanServer mbeanServer;
156
157    private Server server;
158
159    @Override
160    public void start() {
161        server = new Server();
162
163        server.setHandler(handler.getHandler());
164
165        for (ConnectorConfig config : connectors) {
166            server.addConnector(config.getConnector(server));
167        }
168
169        if (mbeanServer != null) {
170            MBeanContainer mbContainer = new MBeanContainer(mbeanServer);
171            server.addBean(mbContainer);
172            server.addBean(Log.getLog());
173        }
174
175        try {
176            server.start();
177        } catch (RuntimeException e) {
178            throw e;
179        } catch (Exception e) {
180            throw new RuntimeException(e);
181        }
182    }
183
184    @Override
185    public void close() {
186        if (server == null) {
187            return;
188        }
189        try {
190            server.stop();
191        } catch (Exception e) {
192            throw new RuntimeException(e);
193        }
194    }
195}
