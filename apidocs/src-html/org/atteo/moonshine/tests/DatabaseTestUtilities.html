002package org.atteo.moonshine.tests;
003
004import javax.xml.bind.annotation.XmlElement;
005import javax.xml.bind.annotation.XmlIDREF;
006import javax.xml.bind.annotation.XmlRootElement;
007
008import org.atteo.moonshine.database.DatabaseService;
009import org.atteo.moonshine.services.ImportBindings;
010import org.atteo.moonshine.services.TopLevelService;
011
012import com.google.inject.AbstractModule;
013import com.google.inject.Module;
014import com.google.inject.Scopes;
015import com.google.inject.matcher.Matchers;
016
017@XmlRootElement(name = "database-tests")
018public class DatabaseTestUtilities extends TopLevelService {
019    @XmlElement
020    @XmlIDREF
021    @ImportBindings
022    private DatabaseService database;
023
024    @Override
025    public Module configure() {
026        return new AbstractModule() {
027            @Override
028            protected void configure() {
029                bind(DatabaseCleaner.class).in(Scopes.SINGLETON);
030
031                FixtureInterceptor interceptor = new FixtureInterceptor();
032                requestInjection(interceptor);
033                bindInterceptor(Matchers.any(), Matchers.annotatedWith(Fixture.class), interceptor);
034            }
035        };
036    }
037}
