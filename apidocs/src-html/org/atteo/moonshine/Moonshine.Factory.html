042/**
043 * Moonshine framework starting class.
044 * <h3>The following operations are performed:</h3>
045 * <p>
046 * <ul>
047 * <li>first, 'Bootstrapping Moonshine' message is logged through SLF4J, this should trigger
048 * logging framework initialization, by default Moonshine is configured to use {@link Logback}
049 * and also provides the default logback.xml file which logs WARN and ERROR messages to the console</li>
050 * <li>shutdown hook is registered, so Moonshine will shutdown cleanly before virtual machine stops</li>
051 * <li>{@link Logging#earlyBootstrap()} is called, the default implementation for Logback redirects
052 * JUL logs through SLF4J and also initializes {@link LevelChangePropagator}</li>
053 * <li>command line parameters are parsed</li>
054 * <li>{@link FileAccessor file accessor} is initialized</li>
055 * <li>{@link Logging#initialize(FileAccessor, Properties)} is called, the default implementation
056 * for Logback loads logback-moonshine.xml which should load final logging configuration, by default Moonshine
057 * contains logback-moonshine.xml file which logs INFO messages to the file in ${logHome}
058 * directory</li>
059 * <li>configuration is loaded from a number of configuration files, by default '/default-config.xml'
060 * classpath resource and ${configHome}/config.xml file, the result is {@link Config} object,</li>
061 * <li>list of configured services is obtained by executing {@link Config#getSubServices()},</li>
062 * <li> {@link Guice} {@link Injector injector} is created based on the modules returned from each Service's
063 * {@link Service#configure()} method,</li>
064 *  <li>{@link Injector#injectMembers members injection} is performed on each service,</li>
065 *  <li>{@link Service#start() start} is executed on each service.</li>
066 * </ul>
067 * </p>
068 *
069 * <h3>Configuration files</h3>
070 * <p>
071 * Two configuration files are searched for. First the classpath is searched for '/default-config.xml'.
072 * The idea is that this file contains default configuration prepared by the application programmer.
073 * Next application configuration directory '${configHome} is searched for config.xml file which should
074 * contain the configuration prepared by the application administrator. If config.xml file does not exist
075 * it is created with the reference to the XSD schema to which the file should conform.
076 * </p>
077 *
078 * <h3>Property resolvers</h3>
079 * <p>
080 * Configuration files are merged and then filtered with a number of predefined {@link PropertyResolver}s.
081 * In the files @code ${name}} placeholder will be replaced in the following ways:
082 * <ul>
083 *   <li>all Java system properties, see {@link SystemPropertyResolver},</li>
084 *   <li>environment variables can be referenced using env prefix, ex ${env.PATH},
085 * see {@link EnvironmentPropertyResolver},</li>
086 *   <li>all elements in the XML configuration file can be referenced using dot to separate tag names,
087 * see {@link XmlPropertyResolver},</li>
088 *   <li>custom properties under {@code <properties>} section in the configuration file,</li>
089 *   <li>properties are resolved recursively, for instance: ${env.${VARNAME}}</li>
090 *   <li>you can add your own custom {@link PropertyResolver}s using
091 * {@link Moonshine.Builder#addPropertyResolver(PropertyResolver)}.</li>
092 * </ul>
093 * Read the description of the {@link Configuration} engine to learn more about merging, filtering
094 * and validating the configuration file.
095 * </p>
096 *
097 * </li>
098 * </ul>
099 * </p>
100 */
101public interface Moonshine extends AutoCloseable {
102    public interface RestrictedBuilder {
103        /**
104         * Enables or disables registering shutdown hook with {@link Runtime#addShutdownHook(Thread)}.
105         */
106        RestrictedBuilder shutdownHook(boolean shutdownHook);
107
108        /**
109         * Sets home directory for default file accessor.
110         */
111        RestrictedBuilder homeDirectory(String homeDirectory);
112
113        /**
114         * Skips default configuration files.
115         * <p>
116         * By default '/default-config.xml' classpath resource and ${configHome}/config.xml configuration
117         * files are read. This option ignores them.
118         * </p>
119         */
120        RestrictedBuilder skipDefaultConfigurationFiles();
121
122        /**
123         * Adds configuration from given resource.
124         */
125        RestrictedBuilder addConfigurationFromResource(String resource);
126
127        /**
128         * Adds configuration from given resource, if it exists.
129         *
130         * <p>
131         * Ignores the resource, if it does not exist.
132         * </p>
133         */
134        RestrictedBuilder addOptionalConfigurationFromResource(String config);
135
136        /**
137         * Adds custom Guice module.
138         */
139        RestrictedBuilder addModule(Module module);
140
141        /**
142         * Adds configuration from given string.
143         */
144        RestrictedBuilder addConfigurationFromString(String string);
145
146        /**
147         * Add property resolver.
148         */
149        RestrictedBuilder addPropertyResolver(PropertyResolver propertyResolver);
150
151        /**
152         * Add read-only config directory.
153         */
154        RestrictedBuilder addConfigDir(String path);
155
156        /**
157         * Add read-only data directory.
158         */
159        RestrictedBuilder addDataDir(String path);
160    }
161
162    public interface Builder extends RestrictedBuilder {
163        /**
164         * Sets application name.
165         */
166        Builder applicationName(String applicationName);
167
168        /**
169         * Sets program arguments.
170         *
171         * <p>
172         * Usually this will be the value of argv parameter from main(String[]) method.
173         * </p>
174         */
175        Builder arguments(String[] arguments);
176
177        /**
178         * Do not register uncaught exception handler.
179         */
180        Builder skipUncaughtExceptionHandler();
181
182        /**
183         * Adds parameter processor.
184         *
185         * <p>
186         * Moonshine parses the parameters using {@link JCommander}. The provided object
187         * will be added to the list of parameter processors with {@link JCommander#addObject(Object)}.
188         * </p>
189         */
190        Builder addParameterProcessor(ParameterProcessor parameterProcessor);
191
192        @Override
193        Builder shutdownHook(boolean shutdownHook);
194
195        /**
196         * Sets logging framework implementation.
197         */
198        Builder loggingFramework(Logging logging);
199
200        @Override
201        Builder homeDirectory(String homeDirectory);
202
203        @Override
204        Builder skipDefaultConfigurationFiles();
205
206        @Override
207        Builder addConfigurationFromResource(String resource);
208
209        @Override
210        Builder addOptionalConfigurationFromResource(String config);
211
212        @Override
213        Builder addModule(Module module);
214
215        @Override
216        Builder addConfigurationFromString(String string);
217
218        @Override
219        Builder addPropertyResolver(PropertyResolver propertyResolver);
220
221        @Override
222        Builder addConfigDir(String path);
223
224        @Override
225        Builder addDataDir(String path);
226
227        /**
228         * Builds Moonshine based on this builder parameters.
229         * <p>
230         * Can return null, if based on provided configuration Moonshine container
231         * is not supposed to be built. For instance, when '--help' command line parameter
232         * is provided Moonshine logs help message and exits immediately.
233         * </p>
234         * @return created Moonshine container, or null if intended behavior is to skip container creation
235         * @throws IOException when configuration could not be accessed
236         * @throws IncorrectConfigurationException when configuration is incorrect
237         */
238        @Nullable
239        Moonshine build() throws MoonshineException, IOException;
240    }
241
242    public static class Factory {
243        public static Builder builder() {
244            return new MoonshineImplementation();
245        }
246
247        public static void logException(MoonshineException e) {
248            Logger logger = LoggerFactory.getLogger("Moonshine");
249
250            if (e instanceof ConfigurationException) {
251                logger.error("Incorrect configuration file: " + e.getMessage());
252                logger.debug("Incorrect configuration file", e);
253            } else if (e instanceof CommandLineParameterException) {
254                logger.error(e.getMessage());
255                logger.debug(e.getMessage(), e);
256            } else {
257                logger.error("Fatal error: " + e.getMessage());
258                logger.debug("Fatal error", e);
259            }
260        }
261    }
262
263    /**
264     * Starts all configured services.
265     */
266    void start();
267
268    /**
269     * Returns the global injector of the Moonshine.
270     * @return Moonshine global injector
271     */
272    Injector getGlobalInjector();
273
274    /**
275     * Stops Moonshine framework.
276     */
277    @Override
278    void close();
279}
