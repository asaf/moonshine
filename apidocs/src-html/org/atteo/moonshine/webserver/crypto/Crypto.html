016package org.atteo.moonshine.webserver.crypto;

018import java.io.File;
019import java.io.FileInputStream;
020import java.io.FileOutputStream;
021import java.io.IOException;
022import java.math.BigInteger;
023import java.security.InvalidKeyException;
024import java.security.KeyManagementException;
025import java.security.KeyPair;
026import java.security.KeyPairGenerator;
027import java.security.KeyStore;
028import java.security.KeyStoreException;
029import java.security.NoSuchAlgorithmException;
030import java.security.NoSuchProviderException;
031import java.security.Provider;
032import java.security.SecureRandom;
033import java.security.SignatureException;
034import java.security.cert.Certificate;
035import java.security.cert.CertificateException;
036import java.security.cert.X509Certificate;
037import java.util.Date;
038
039import javax.net.ssl.SSLContext;
040import javax.net.ssl.TrustManager;
041import javax.net.ssl.X509TrustManager;
042
043import org.bouncycastle.asn1.x500.X500Name;
044import org.bouncycastle.asn1.x500.X500NameBuilder;
045import org.bouncycastle.asn1.x500.style.BCStyle;
046import org.bouncycastle.cert.X509v3CertificateBuilder;
047import org.bouncycastle.cert.jcajce.JcaX509CertificateConverter;
048import org.bouncycastle.cert.jcajce.JcaX509v3CertificateBuilder;
049import org.bouncycastle.jce.provider.BouncyCastleProvider;
050import org.bouncycastle.operator.ContentSigner;
051import org.bouncycastle.operator.OperatorCreationException;
052import org.bouncycastle.operator.jcajce.JcaContentSignerBuilder;
053
054public class Crypto {
055    public static void createSelfSignedCertificate(File keystore, String alias,
056            String keystorePassword) {
057        try {
058            Provider bouncyCastleProvider = new BouncyCastleProvider();
059
060            KeyPairGenerator keyPairGenerator = KeyPairGenerator.getInstance("RSA", bouncyCastleProvider);
061            keyPairGenerator.initialize(1024, new SecureRandom());
062            KeyPair keyPair = keyPairGenerator.generateKeyPair();
063
064            // Generate self-signed certificate
065            X500NameBuilder nameBuilder = new X500NameBuilder(BCStyle.INSTANCE);
066            nameBuilder.addRDN(BCStyle.CN, "localhost");
067            X500Name name = nameBuilder.build();
068
069            Date notBefore = new Date(System.currentTimeMillis() - 24 * 60 * 60 * 1000);
070            Date notAfter = new Date(System.currentTimeMillis() + 10 * 365 * 24 * 60 * 60 * 1000);
071            BigInteger serial = BigInteger.valueOf(System.currentTimeMillis());
072
073            X509v3CertificateBuilder certGen = new JcaX509v3CertificateBuilder(name,
074                    serial, notBefore, notAfter, name, keyPair.getPublic());
075            ContentSigner sigGen = new JcaContentSignerBuilder("SHA256WithRSAEncryption")
076                    .setProvider(bouncyCastleProvider).build(keyPair.getPrivate());
077            X509Certificate cert = new JcaX509CertificateConverter().setProvider(bouncyCastleProvider)
078                    .getCertificate(certGen.build(sigGen));
079            cert.checkValidity(new Date());
080            cert.verify(cert.getPublicKey());
081
082            // Save to keystore
083            KeyStore store = KeyStore.getInstance("JKS");
084            if (keystore.exists()) {
085                try (FileInputStream fis = new FileInputStream(keystore)) {
086                    store.load(fis, keystorePassword.toCharArray());
087                }
088            } else {
089                store.load(null);
090            }
091            store.setKeyEntry(alias, keyPair.getPrivate(), keystorePassword.toCharArray(),
092                    new Certificate[] { cert });
093            try (FileOutputStream fos = new FileOutputStream(keystore)) {
094                store.store(fos, keystorePassword.toCharArray());
095            }
096        } catch (NoSuchAlgorithmException | OperatorCreationException | CertificateException
097                | InvalidKeyException | NoSuchProviderException | SignatureException | KeyStoreException
098                | IOException e) {
099            throw new RuntimeException(e);
100        }
101    }
102
103    public static SSLContext createNoValidationSSLContext() {
104        try {
105            TrustManager[] trustAllCerts = new TrustManager[] {
106                new X509TrustManager() {
107                    @Override
108                    public java.security.cert.X509Certificate[] getAcceptedIssuers() {
109                        return null;
110                    }
111
112                    @Override
113                    public void checkClientTrusted(X509Certificate[] certs, String authType) {
114                    }
115
116                    @Override
117                    public void checkServerTrusted(X509Certificate[] certs, String authType) {
118                    }
119                }
120            };
121
122            SSLContext sslContext = SSLContext.getInstance("SSL");
123            sslContext.init(null, trustAllCerts, new java.security.SecureRandom());
124            return sslContext;
125        } catch (KeyManagementException | NoSuchAlgorithmException e) {
126            throw new RuntimeException("Cannot create SSLContext which ignores certificates", e);
127        }
128    }
129}
