016package org.atteo.moonshine.services;
017
018import javax.annotation.Nonnull;
019import javax.inject.Inject;
020
021import org.atteo.evo.classindex.IndexSubclasses;
022
023import com.google.inject.Guice;
024import com.google.inject.Injector;
025import com.google.inject.Module;
026import com.google.inject.Provider;
027
028/**
029 * Fundamental Moonshine component encapsulating some functionality.
030 * <p>
031 * Service can {@link #configure() register bindings}, can be repeatedly {@link #start() started}
032 * and {@link #stop() stopped} and finally {@link #close() closed}.
033 * </p>
034 * <p>
035 * Services are instantiated by {@link Services} according to the provided configuration.
036 * Most of the logic which configures the Service should be put in {@link #configure()} method,
037 * so it can be lazily initialized by {@link Guice} which knows about dependencies between services.
038 * </p>
039 * <p>
040 * {@link #configure()} is executed only once before creating {@link Injector Guice injector} and can be used
041 * to configure it. {@link #close()} is also executed only once, just before destroying the injector.
042 * {@link #start()} and {@link #stop()} methods should start and stop any functionality which this service
043 * provides to the outside world.
044 * </p>
045 */
046@IndexSubclasses(storeJavadoc = true)
047public interface Service extends AutoCloseable {
048    String getId();
049
050    /**
051     * Returns {@link Module} which specifies this service dependencies and what it provides
052     * to other services.
053     *
054     * @return Guice module
055     */
056    Module configure();
057
058    /**
059     * Starts this service.
060     *
061     * <p>
062     * All fields marked with {@link Inject} will be injected before execution of this method.
063     * </p>
064     * <p>
065     * This method should be used to start functionalities which are used outside this application.
066     * Functionality that is exported to other services should be started in {@link #configure()} method
067     * or from {@link Provider} to correctly handle dependencies between services.
068     * </p>
069     */
070    void start();
071
072    /**
073     * Stops this service.
074     *
075     * <p>
076     * Service can still be started later using {@link #start()} method.
077     * </p>
078     */
079    void stop();
080
081    /**
082     * Closes this services, it won't be started again.
083     */
084    @Override
085    void close();
086
087    /**
088     * Returns a list of direct sub-services this service contains.
089     *
090     * <p>
091     * Although usually the list will come from unmarshalling the XML file,
092     * there is nothing which forbids the service from generating the list on the fly.
093     * </p>
094     * <p>
095     * Only direct sub-services should be returned. Moonshine will on its own ask
096     * returned sub-services for their sub-services.
097     * </p>
098     * @return list of services
099     */
100    @Nonnull
101    Iterable<? extends Service> getSubServices();
102}
