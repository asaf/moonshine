029import com.google.common.base.Strings;

032import com.google.inject.BindingAnnotation;
033import com.google.inject.Key;

036import com.google.inject.name.Names;

042public class ServiceModuleRewriter {
043    public static List<Element> annotateExposedWithId(final List<Element> elements, final Service service) {
044        final boolean singleton = ReflectionTools.isSingleton(service.getClass());
045        return Elements.getElements(new Module() {
046            @Override
047            public void configure(final Binder binder) {
048                final PrivateBinder privateBinder = binder.newPrivateBinder();
049                privateBinder.requestInjection(service);
050
051                for (Element element : elements) {
052                    element.acceptVisitor(new DefaultElementVisitor<Void>() {
053                        @Override
054                        public Void visit(PrivateElements privateElements) {
055                            PrivateBinder privateBinder = binder.newPrivateBinder().withSource(
056                                    privateElements.getSource());
057
058                            annotateExposedWithId(privateBinder, privateElements, service);
059                            return null;
060                        }
061
062                        @Override
063                        public <T> Void visit(Binding<T> binding) {
064                            binding.applyTo(privateBinder);
065                            Key<T> oldKey = binding.getKey();
066                            if (!singleton && !Strings.isNullOrEmpty(service.getId())) {
067                                bindKey(privateBinder, oldKey, Names.named(service.getId()));
068                            } else {
069                                privateBinder.expose(oldKey);
070                            }
071                            return null;
072                        }
073
074                        @Override
075                        protected Void visitOther(Element element) {
076                            element.applyTo(binder);
077                            return null;
078                        }
079                    });
080                }
081            }
082        });
083    }
084
085    private static <T> void bindKey(PrivateBinder binder, Key<T> oldKey, Annotation annotation) {
086        Key<T> newKey = Key.get(oldKey.getTypeLiteral(), annotation);
087        binder.bind(newKey).to(oldKey);
088        binder.expose(newKey);
089    }
090
091    private static void annotateExposedWithId(final PrivateBinder binder,
092            final PrivateElements elements, final Service service) {
093
094        final boolean singleton = ReflectionTools.isSingleton(service.getClass());
095
096        // rewrite all bindings
097        for (Element element : elements.getElements()) {
098            element.applyTo(binder);
099        }
100
101        // rewrite all exposed bindings to include id
102        for (Key<?> oldKey : elements.getExposedKeys()) {
103            if (!singleton && !Strings.isNullOrEmpty(service.getId())) {
104                bindKey(binder, oldKey, Names.named(service.getId()));
105            } else {
106                binder.expose(oldKey);
107            }
108        }
109    }
110
111    /**
112     * Create a module with all provided elements and also with bindings imported
113     * with {@link ImportBindings} annotation.
114     * @param elements elements to include in the module
115     * @param service service to scan for {@link ImportBindings} annotation
116     * @param serviceElements map with elements for all services
117     * @param hints list where hints will be stored
118     * @return provided elements with bindings imported from given service
119     */
120    public static List<Element> importBindings(final List<Element> elements, final Service service,
121            final Map<Service, List<Element>> serviceElements, final List<String> hints) {
122        return Elements.getElements(new Module() {
123            @Override
124            public void configure(final Binder binder) {
125                for (Element element : elements) {
126                    element.acceptVisitor(new DefaultElementVisitor<Void>() {
127                        @Override
128                        public Void visit(PrivateElements privateElements) {
129                            PrivateBinder privateBinder = binder.newPrivateBinder().withSource(
130                                    privateElements.getSource());
131
132                            importBindings(privateBinder, privateElements, service, serviceElements, hints);
133                            return null;
134                        }
135
136                        @Override
137                        protected Void visitOther(Element element) {
138                            // copy all elements
139                            element.applyTo(binder);
140                            return null;
141                        }
142                    });
143                }
144            }
145        });
146    }
147
148    /**
149     * Find annotation on the field which is itself annotated with {@link BindingAnnotation}.
150     * @param field field with {@link Field#setAccessible(boolean)} already called
151     * @return binding annotation or null, if not found
152     */
153    private static Annotation findBindingAnnotation(Field field) {
154        Annotation result = null;
155        for (Annotation annotation : field.getAnnotations()) {
156            if (annotation.annotationType().isAnnotationPresent(BindingAnnotation.class)) {
157                if (result == null) {
158                    result = annotation;
159                } else {
160                    throw new RuntimeException("More than one binding annotation specified");
161                }
162            }
163        }
164        return result;
165    }
166
167    private static void importBindings(final PrivateBinder binder, PrivateElements elements, Service service,
168            Map<Service, List<Element>> serviceElements, List<String> hints) {
169
170        // copy all elements
171        for (Element element : elements.getElements()) {
172            element.applyTo(binder);
173        }
174        for (Key<?> key : elements.getExposedKeys()) {
175            binder.expose(key);
176        }
177
178        for (final Field field : service.getClass().getDeclaredFields()) {
179            if (!field.isAnnotationPresent(ImportBindings.class)) {
180                continue;
181            }
182
183            if (!Service.class.isAssignableFrom(field.getType())) {
184                throw new RuntimeException("@" + ImportBindings.class.getSimpleName() + " annotation can only"
185                        + " be specified on a field of type " + Service.class.getSimpleName());
186            }
187
188            AccessController.doPrivileged(new PrivilegedAction<Void>() {
189                @Override
190                public Void run() {
191                    field.setAccessible(true);
192                    return null;
193                }
194            });
195            Service importedService;
196            try {
197                importedService = (Service) field.get(service);
198            } catch (IllegalAccessException| IllegalArgumentException e) {
199                throw new RuntimeException("Cannot access field", e);
200            }
201
202            List<Element> importedElements;
203
204            if (importedService != null) {
205                importedElements = serviceElements.get(importedService);
206                if (importedElements == null) {
207                    throw new RuntimeException("Imported service does not specify any module");
208                }
209            } else {
210                importedElements = findDefaultServiceElements(serviceElements, field.getType());
211                if (importedElements == null) {
212                    hints.add("Service '" + service.toString() + "' depends on '" + field.getType().getSimpleName()
213                            + "' which is not available");
214                    importedElements = Collections.emptyList();
215                }
216            }
217
218            for (Element element : importedElements) {
219                element.acceptVisitor(new DefaultElementVisitor<Void>() {
220                    private <T> void bindKey(Key<T> key) {
221                        Key<T> sourceKey;
222
223                        Annotation annotation = findBindingAnnotation(field);
224                        if (annotation == null) {
225                            sourceKey = Key.get(key.getTypeLiteral());
226                        } else {
227                            sourceKey = Key.get(key.getTypeLiteral(), annotation);
228                        }
229
230                        if (!sourceKey.equals(key)) {
231                            binder.bind(sourceKey).to(key);
232                        }
233                    }
234
235                    @Override
236                    public Void visit(PrivateElements privateElements) {
237                        for (Key<?> key : privateElements.getExposedKeys()) {
238                            bindKey(key);
239                        }
240                        return null;
241                    }
242
243                    @Override
244                    public <T> Void visit(Binding<T> binding) {
245                        bindKey(binding.getKey());
246                        return null;
247                    }
248                });
249            }
250        }
251    }
252
253    private static List<Element> findDefaultServiceElements(Map<Service, List<Element>> serviceElements,
254            Class<?> type) {
255
256        List<Element> result = null;
257        for (Map.Entry<Service, List<Element>> entry : serviceElements.entrySet()) {
258            Service service = entry.getKey();
259            List<Element> elements = entry.getValue();
260
261            if (type.isAssignableFrom(service.getClass())) {
262                if (result != null) {
263                    throw new RuntimeException("Could not find unique service of type: " + type);
264                }
265                result = elements;
266            }
267        }
268        return result;
269    }
