023import java.util.List;
024import java.util.Map;
025import java.util.Set;
026
027import org.atteo.evo.classindex.ClassIndex;
028import org.atteo.evo.urlhandlers.UrlHandlers;
029import org.atteo.moonshine.ConfigurationException;
030import org.atteo.moonshine.injection.InjectMembersModule;
031import org.atteo.moonshine.services.internal.DuplicateDetectionWrapper;
032import org.atteo.moonshine.services.internal.ReflectionTools;
033import org.atteo.moonshine.services.internal.ServiceModuleRewriter;

035import org.slf4j.LoggerFactory;

037import com.google.common.base.Strings;
038import com.google.inject.CreationException;
039import com.google.inject.Guice;
040import com.google.inject.Injector;
041import com.google.inject.Module;
042import com.google.inject.servlet.ServletModule;
043import com.google.inject.spi.Element;
044import com.google.inject.spi.Elements;
045
046public class ServicesImplementation implements Services, Services.Builder {
047    private final Logger logger = LoggerFactory.getLogger("Moonshine");
048    private final List<Module> extraModules = new ArrayList<>();
049
050    private Injector injector;
051    private ServicesConfig config;
052    private final List<Service> startedServices = new ArrayList<>();
053    private final Map<Service, String> serviceNameMap = new IdentityHashMap<>();
054    private final Map<Service, List<com.google.inject.spi.Element>> serviceElements = new LinkedHashMap<>();
055
056    public ServicesImplementation() {
057        UrlHandlers.registerAnnotatedHandlers();
058    }
059
060    @Override
061    public Builder addModule(Module module) {
062        extraModules.add(module);
063        return this;
064    }
065
066    @Override
067    public Builder configuration(ServicesConfig config) {
068        this.config = config;
069        return this;
070    }
071
072    @Override
073    public Services build() throws ConfigurationException {
074        createInjector();
075        return this;
076    }
077
078    private Injector buildInjector() {
079        List<Module> modules = new ArrayList<>();
080        DuplicateDetectionWrapper duplicateDetection = new DuplicateDetectionWrapper();
081
082        // Use ServletModule specifically so @RequestScoped annotation will be always bound
083        Module servletsModule = duplicateDetection.wrap(new ServletModule() {
084            @Override
085            public void configureServlets() {
086                binder().requireExplicitBindings();
087            }
088        });
089
090        // important magic below:
091        // Every ServletModule instance tries to install InternalServletModule. The trick is used, because Guice
092        // installs modules only the first time and ignores any subsequent execution of install method
093        // with the same module (by comparing them using equals() method).
094        // We need to make sure InternalServletModule is installed in the top level module,
095        // because when it is installed from some private module it doesn't have an access to all
096        // registered servlets and filters.
097        //
098        // The line below makes sure first instance of ServletModule is rewritten in global scope
099        modules.add(Elements.getModule(Elements.getElements(servletsModule)));
100
101        for (Module module : extraModules) {
102            modules.add(duplicateDetection.wrap(module));
103        }
104
105        for (Service service : config.getSubServices()) {
106            logger.info("Configuring: {}", serviceNameMap.get(service));
107            Module module = service.configure();
108            if (module != null) {
109                serviceElements.put(service, Elements.getElements(duplicateDetection.wrap(module)));
110            } else {
111                serviceElements.put(service, Collections.<com.google.inject.spi.Element>emptyList());
112            }
113        }
114
115        for (Map.Entry<Service, List<com.google.inject.spi.Element>> entry : serviceElements.entrySet()) {
116            Service service = entry.getKey();
117            List<com.google.inject.spi.Element> elements = entry.getValue();
118
119            serviceElements.put(service, ServiceModuleRewriter.annotateExposedWithId(elements, service));
120        }
121
122        List<String> hints = new ArrayList<>();
123
124        for (Map.Entry<Service, List<com.google.inject.spi.Element>> entry : serviceElements.entrySet()) {
125            Service service = entry.getKey();
126            List<com.google.inject.spi.Element> elements = entry.getValue();
127
128            serviceElements.put(service, ServiceModuleRewriter.importBindings(elements, service, serviceElements,
129                    hints));
130        }
131
132        for (Map.Entry<Service, List<com.google.inject.spi.Element>> entry : serviceElements.entrySet()) {
133            List<com.google.inject.spi.Element> elements = entry.getValue();
134            modules.add(Elements.getModule(elements));
135        }
136
137        modules.add(new InjectMembersModule());
138
139        try {
140            return Guice.createInjector(modules);
141        } catch (CreationException e) {
142            if (!hints.isEmpty()) {
143                logger.warn("Problem detected while creating Guice injector, possible causes:");
144                for (String hint : hints) {
145                    logger.warn(" -> " + hint);
146                }
147            }
148            throw e;
149        }
150    }
151
152    private void createInjector() throws ConfigurationException {
153        logger.info("Building Guice injector hierarchy");
154
155        if (config == null) {
156            config = new ServicesConfig();
157        }
158
159        buildServiceNameMap(config.getSubServices());
160        verifySingletonServicesAreUnique(config.getSubServices());
161        injector = buildInjector();
162    }
163
164    @Override
165    public Injector getGlobalInjector() {
166        return injector;
167    }
168
169    @Override
170    public void start() {
171        logger.info("Starting services");
172        for (Service service : config.getSubServices()) {
173            if (logger.isInfoEnabled()
174                    && ReflectionTools.isMethodOverriden(service.getClass(), Service.class, "start")) {
175                logger.info("Starting: {}", serviceNameMap.get(service));
176            }
177            startedServices.add(service);
178            service.start();
179        }
180        logger.info("All services started");
181    }
182
183    @Override
184    public void stop() {
185        if (config == null) {
186            return;
187        }
188
189        for (Service service : startedServices) {
190            String name = serviceNameMap.get(service);
191            if (name == null) {
192                name = service.getClass().getSimpleName();
193            }
194            logger.info("Stopping: {}", name);
195            service.stop();
196        }
197    }
198
199    @Override
200    public void close() {
201        stop();
202        for (Service service : config.getSubServices()) {
203            service.close();
204        }
205        if (logger != null) {
206            logger.info("All services stopped");
207        }
208        injector = null;
209    }
210
211    @Override
212    public Map<Service, List<Element>> getServiceElements() {
213        return serviceElements;
214    }
215
216    private void verifySingletonServicesAreUnique(List<Service> services) throws ConfigurationException {
217        Set<Class<?>> set = new HashSet<>();
218        for (Service service : services) {
219            Class<?> klass = service.getClass();
220            if (ReflectionTools.isSingleton(klass)) {
221                if (set.contains(klass)) {
222                    throw new ConfigurationException("Service '" + klass.getCanonicalName() + "' is marked"
223                            + " as singleton, but is declared more than once in configuration file");
224                }
225                set.add(klass);
226
227                if (!Strings.isNullOrEmpty(service.getId())) {
228                    throw new ConfigurationException("Service '" + klass.getCanonicalName() + "' is marked"
229                            + " as singleton, but has an id specified");
230                }
231            }
232        }
233    }
234
235    private void buildServiceNameMap(List<Service> services) {
236        for (Service service : services) {
237            StringBuilder builder = new StringBuilder();
238
239            if (service.getId() != null) {
240                builder.append("\"");
241                builder.append(service.getId());
242                builder.append("\" ");
243            }
244
245            String summary = ClassIndex.getClassSummary(service.getClass());
246            builder.append(service.getClass().getSimpleName());
247            if (summary != null) {
248                builder.append(" (");
249                builder.append(summary);
250                builder.append(")");
251            }
252            serviceNameMap.put(service, builder.toString());
253        }
254    }
