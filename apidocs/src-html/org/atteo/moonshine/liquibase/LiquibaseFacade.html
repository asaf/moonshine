014package org.atteo.moonshine.liquibase;
015
016import java.sql.SQLException;
017
018import javax.inject.Inject;
019import javax.sql.DataSource;
020
021import liquibase.Liquibase;
022import liquibase.database.DatabaseConnection;
023import liquibase.database.jvm.JdbcConnection;
024import liquibase.exception.DatabaseException;
025import liquibase.exception.LiquibaseException;
026import liquibase.resource.ClassLoaderResourceAccessor;
027import liquibase.resource.ResourceAccessor;
028
029/**
030 * Liquibase facade for migrations execution.
031 * <p>
032 * Usage:
033 * <pre>
034 * new LiquibaseFacade(datasource).migrate("/migrations/migration01.xml");
035 * </pre>
036 * </p>
037 */
038public class LiquibaseFacade {
039    private static final String BEFORE_LAST_UPDATE = "BEFORE_LAST_UPDATE";
040
041    private final DataSource dataSource;
042
043    @Inject
044    public LiquibaseFacade(DataSource dataSource) {
045        this.dataSource = dataSource;
046    }
047
048    /**
049     * Migrate using given migration.
050     * @param changelog resource with the migration
051     */
052    public void migrate(String changelog) {
053        migrate(changelog, null);
054    }
055
056    /**
057     * Migrate using given migration.
058     * @param changelog resource with the migration
059     * @param contexts contexts, see {@link Liquibase#update(String)}.
060     */
061    public void migrate(String changelog, String contexts) {
062        ResourceAccessor resourceAccessor = new ClassLoaderResourceAccessor();
063
064        DatabaseConnection databaseConnection = null;
065        changelog = normalizeName(changelog);
066
067        try {
068            databaseConnection = new JdbcConnection(dataSource.getConnection());
069            Liquibase liquibase = new Liquibase(changelog, resourceAccessor, databaseConnection);
070            liquibase.tag(BEFORE_LAST_UPDATE);
071            liquibase.update(contexts);
072        } catch (LiquibaseException | SQLException e) {
073            throw new RuntimeException(e);
074        } finally {
075            try {
076                if (databaseConnection != null && !databaseConnection.isClosed())
077                    databaseConnection.close();
078            } catch (DatabaseException e) {
079                throw new RuntimeException(e);
080            }
081        }
082    }
083
084    /**
085     * Rollbacks last update.
086     * @param changelog resource with the migration
087     */
088    public void rollbackLastUpdate(String changelog) {
089        rollback(changelog, null, BEFORE_LAST_UPDATE);
090    }
091
092    /**
093     * Rollbacks last update.
094     * @param changelog resource with the migration
095     * @param contexts contexts, see {@link Liquibase#update(String)}.
096     */
097    public void rollbackLastUpdate(String changelog, String contexts) {
098        rollback(changelog, contexts, BEFORE_LAST_UPDATE) ;
099    }
100
101    /**
102     * Rollbacks given changelog to given tag.
103     * @param changelog resource with the migration
104     * @param contexts contexts, see {@link Liquibase#update(String)}.
105     * @param tag tag to rollback to
106     */
107    private void rollback(String changelog, String contexts, String tag) {
108        changelog = normalizeName(changelog);
109
110        ResourceAccessor resourceAccessor = new ClassLoaderResourceAccessor();
111
112        DatabaseConnection databaseConnection = null;
113
114        try {
115            databaseConnection = new JdbcConnection(dataSource.getConnection());
116            Liquibase liquibase = new Liquibase(changelog, resourceAccessor, databaseConnection);
117            liquibase.rollback(tag, contexts);
118        } catch (LiquibaseException | SQLException e) {
119            throw new RuntimeException(e);
120        } finally {
121            try {
122                if (databaseConnection != null && !databaseConnection.isClosed()) {
123                    databaseConnection.close();
124                }
125            } catch (DatabaseException e) {
126                throw new RuntimeException(e);
127            }
128        }
129    }
130
131    public void dropAll() {
132        ResourceAccessor resourceAccessor = new ClassLoaderResourceAccessor();
133
134        DatabaseConnection databaseConnection = null;
135
136        try {
137            databaseConnection = new JdbcConnection(dataSource.getConnection());
138            Liquibase liquibase = new Liquibase(null, resourceAccessor, databaseConnection);
139            liquibase.dropAll();
140        } catch (LiquibaseException | SQLException e) {
141            throw new RuntimeException(e);
142        } finally {
143            try {
144                if (databaseConnection != null && !databaseConnection.isClosed())
145                    databaseConnection.close();
146            } catch (DatabaseException e) {
147                throw new RuntimeException(e);
148            }
149        }
150    }
151
152    private String normalizeName(String changelog) {
153        if (changelog.startsWith("/") ){
154            changelog = changelog.substring(1);
155        }
156        return changelog;
157    }
158}
